Chart A

rest_inspec %>% 
  separate(inspection_date, 
           into = c("year", "month", "date"), 
           sep = "\\-", 
           remove = FALSE) %>% 
  filter(grade %in% c("A", "B", "C"),
         lubridate::year(inspection_date) %in% 2013:2017,
         str_detect(inspection_type, "Initial") == FALSE,
         score > 0) %>% 
  group_by(year) %>% 
  mutate(
    count = n()
  ) %>% 
  filter(grade == "A") %>% 
  mutate(
    count_a = n()) %>% 
  distinct(count, count_a) %>% 
  summarize(
    a_rate = round(count_a/count*100, digits = 2)
  ) %>% 
  plot_ly(x = ~year, y = ~a_rate, type = "scatter", mode = "marker") %>% 
  layout(xaxis =
           list(
             title = "Year"
           ),
         yaxis =
           list(
             title = "'A' Grades as a Percent of Grades"
           ))

Chart B

rest_inspec %>% 
  filter(!is.na(violation_code)) %>% 
  group_by(dba) %>% 
  summarize(
    count = n()) %>%
  filter(count > 500) %>% 
  mutate(dba =
           as.factor(dba),
         dba = 
           fct_reorder(dba, count)) %>% 
  plot_ly(x = ~dba, y = ~count, type = "bar") %>% 
  layout(xaxis =
           list(
             title = "Business"
           ),
         yaxis =
           list(
             title = "Number of Violations"
           ))

Chart C

rest_inspec %>% 
  group_by(violation_code, year(inspection_date)) %>% 
  summarize(
    count = n()) %>% 
  mutate(violation_code =
           as.factor(violation_code),
         violation_code =
           fct_reorder(violation_code, desc(count))) %>% 
  top_n(violation_code, 10, count) %>% 
  plot_ly(y = ~count, color = ~violation_code, type = "box") %>% 
  layout(xaxis =
           list(
             title = "Year"
           ),
         yaxis =
           list(
             title = "Types of Violations"
           ))
